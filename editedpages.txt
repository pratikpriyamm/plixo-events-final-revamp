        // DYNAMIC SLIDE LOGIC
        if (hangingMenu) {
            if (isActive) {
                // 1. Get the actual height of the open dropdown
                // We assume max-height allows it to fully expand, so scrollHeight gives real height
                const exactHeight = navMenu.scrollHeight;
                
                // 2. Move hanging menu down by that EXACT amount
                hangingMenu.style.transform = `translateY(${exactHeight}px)`;
            } else {
                // 3. Reset to 0 when closed
                hangingMenu.style.transform = 'translateY(0)';
            }
        }
    }

    function closeMenu() {
        navMenu.classList.remove('active');
        menuBtn.classList.remove('open');
        
        // Reset hanging menu
        if (hangingMenu) {
            hangingMenu.style.transform = 'translateY(0)';
        }
    }

    // Event Listeners
    menuBtn.addEventListener('click', (e) => {
        e.stopPropagation();
        toggleMenu();
    });

    navLinks.forEach(link => {
        link.addEventListener('click', closeMenu);
    });

    document.addEventListener('click', (e) => {
        if (navMenu.classList.contains('active')) {
            if (!navMenu.contains(e.target) && !menuBtn.contains(e.target)) {
                closeMenu();
            }
        }
    });

    /* -----------------------------------------------
       2. Scroll Animation (Intersection Observer)
    ----------------------------------------------- */
    const observerOptions = {
        root: null, // use the viewport
        threshold: 0.1, // trigger when 10% of element is visible
        rootMargin: "0px"
    };

    const scrollObserver = new IntersectionObserver((entries, observer) => {
        entries.forEach(entry => {
            if (entry.isIntersecting) {
                entry.target.classList.add('visible');
                observer.unobserve(entry.target); // Only animate once
            }
        });
    }, observerOptions);

    // Target all elements with the fade class
    const scrollElements = document.querySelectorAll('.fade-on-scroll');
    scrollElements.forEach(el => scrollObserver.observe(el));

});

/* -----------------------------------------------
       3. FAST SCROLL TO TOP (Logo Click)
    ----------------------------------------------- */
    const logoLink = document.querySelector('.logo-container');

    logoLink.addEventListener('click', (e) => {
        e.preventDefault(); // Stop the default instant jump
        
        const duration = 500; // Duration in milliseconds (0.5s = Very Fast)
        const startY = window.scrollY;
        const startTime = performance.now();

        function scrollAnimation(currentTime) {
            const timeElapsed = currentTime - startTime;
            const progress = Math.min(timeElapsed / duration, 1);

            // Easing function (easeOutQuad) for a natural but fast feel
            // It starts fast and slows down slightly at the very end
            const ease = 1 - (1 - progress) * (1 - progress);

            window.scrollTo(0, startY * (1 - ease));

            if (progress < 1) {
                requestAnimationFrame(scrollAnimation);
            }
        }

        requestAnimationFrame(scrollAnimation);
    });

    /* -----------------------------------------------
       READ MORE / READ LESS LOGIC (Fixed)
    ----------------------------------------------- */
    const readMoreBtn = document.getElementById('read-more-btn');

                aboutWrapper.classList.remove('collapsed');
                readMoreBtn.textContent = "Read Less";
                
                // Hide Fade
                if(fadeOverlay) fadeOverlay.style.opacity = '0';
                
            } else {
                // COLLAPSE
                aboutWrapper.classList.add('collapsed');
                readMoreBtn.textContent = "Read More";
                
                // Show Fade
                if(fadeOverlay) fadeOverlay.style.opacity = '1';
                
                // Smooth scroll back to top of section so user isn't lost
                document.getElementById('about').scrollIntoView({ behavior: 'smooth' });
            }
        });
    }

    /* -----------------------------------------------
       4. HANGING MENU HOME BUTTON (Fast Scroll)
    ----------------------------------------------- */
    // Select the specific home button inside the hanging menu
    const menuHomeBtn = document.querySelector('.hanging-menu a[href="#home"]');

    if (menuHomeBtn) {
        menuHomeBtn.addEventListener('click', (e) => {
            e.preventDefault(); // Stop default jump
            
            const duration = 500; // 0.5 seconds (Very Fast)
            const startY = window.scrollY;
            const startTime = performance.now();

            function scrollAnimation(currentTime) {
                const timeElapsed = currentTime - startTime;
                const progress = Math.min(timeElapsed / duration, 1);
                
                // Natural easing
                const ease = 1 - (1 - progress) * (1 - progress);

                window.scrollTo(0, startY * (1 - ease));

                if (progress < 1) {
                    requestAnimationFrame(scrollAnimation);
                }
            }

            requestAnimationFrame(scrollAnimation);
        });
    }

   SERVICE BLOCKS (Slide Up Logic)
----------------------------------------------- */
function toggleBlock(clickedBlock) {
    // 1. Get all blocks
    const allBlocks = document.querySelectorAll('.service-block');
    
    // 2. Check if this one is already active
    const isActive = clickedBlock.classList.contains('active');
    
    // 3. Reset ALL blocks (Close any open ones)
    allBlocks.forEach(block => {
        block.classList.remove('active');
    });
    
    // 4. If it wasn't active before, make it active now
    if (!isActive) {
        clickedBlock.classList.add('active');
    }
}
